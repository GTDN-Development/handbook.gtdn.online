## Usage

Use the Header and Footer components in your layout:

```tsx title="app/layout.tsx"
import { Header } from "@/components/ui/header";
import { Footer } from "@/components/ui/footer";

export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html>
      <body>
        <Header />
        <main>{children}</main>
        <Footer />
      </body>
    </html>
  );
}
```

## Customization

### Updating Navigation Links

Edit `src/config/nav-links.ts` to customize your navigation:

```tsx
export const primaryNavLinks = {
  home: { name: "Home", href: "/" },
  // Add dropdown menu
  products: {
    name: "Products",
    items: [
      { name: "Product 1", href: "/products/1" },
      { name: "Product 2", href: "/products/2" },
    ],
  },
  // Add simple link
  pricing: { name: "Pricing", href: "/pricing" },
};
```

### Updating Social Media

Edit `src/config/social-media-links.ts`:

```tsx
export const socialMediaLinks = {
  twitter: {
    name: "Twitter",
    href: "https://twitter.com/yourhandle",
    icon: TwitterXIcon,
  },
  // Add more social links...
};
```
