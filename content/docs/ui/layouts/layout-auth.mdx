## Overview

The Layout Auth component provides a clean, centered layout specifically designed for authentication pages. It creates a full-height viewport with vertically and horizontally centered content, perfect for login forms, registration pages, and other auth-related interfaces.

## Usage

Use this layout for authentication pages in your app:

```tsx title="app/login/page.tsx"
import { LayoutAuth } from "@/components/layout/layout-auth";
import { LoginForm } from "@/components/auth/login-form";

export default function LoginPage() {
  return (
    <LayoutAuth>
      <LoginForm />
    </LayoutAuth>
  );
}
```

```tsx title="app/register/page.tsx"
import { LayoutAuth } from "@/components/layout/layout-auth";
import { RegisterForm } from "@/components/auth/register-form";

export default function RegisterPage() {
  return (
    <LayoutAuth>
      <RegisterForm />
    </LayoutAuth>
  );
}
```

  children,
  ...props
