## Usage

```tsx
import { FloatingBar } from "@/components/ui/floating-bar";
```

```tsx
<FloatingBar className="border-b bg-background/95 backdrop-blur">
  <nav className=" flex items-center justify-between py-4">
    <div>Logo</div>
    <div>Navigation</div>
  </nav>
</FloatingBar>
```
