This component is commonly used to build navigation menus, headers, footers, and sidebars. It provides consistent active state styling and proper handling of external links across your entire application.

## Installation

**Create a source file**
```bash
touch src/components/ui/nav-link.tsx
```

Use one of the following examples

## Examples

### Next.js standard

A basic navigation link for standard Next.js applications. Uses `usePathname` to detect the current route and automatically handles external links with proper security attributes.

```tsx
"use client";

import { Link, type LinkProps } from "../ui/link";
import { usePathname } from "next/navigation";
import { ArrowUpRightIcon } from "lucide-react";

export type NavLinkProps = LinkProps & {
  showExternalIcon?: boolean;
  matchNested?: boolean;
};

export function NavLink({
  children,
  href,
  target,
  rel,
  showExternalIcon = false,
  matchNested = false,
  ...props
}: NavLinkProps) {
  const pathname = usePathname();
  const isExternal = typeof href === "string" && href.startsWith("http");
  const hrefString = typeof href === "string" ? href : (href.pathname ?? "");
  const isCurrent = matchNested
    ? pathname === hrefString || pathname.startsWith(`${hrefString}/`)
    : pathname === hrefString;

  return (
    <Link
      {...props}
      href={href}
      aria-current={isCurrent ? "page" : undefined}
      data-current={isCurrent ? "true" : undefined}
      data-external={isExternal ? "true" : undefined}
      target={isExternal ? target || "_blank" : target}
      rel={isExternal ? rel || "noopener noreferrer" : rel}
    >
      {children}
      {isExternal && showExternalIcon && (
        <ArrowUpRightIcon aria-hidden="true" className="ml-1 inline size-[1em] opacity-50" />
      )}
    </Link>
  );
}
```

### With next-intl

For internationalized applications using [next-intl](https://next-intl.dev). Uses `useSelectedLayoutSegment` for accurate active state detection across localized routes.

> **Note:** This example uses `Link` from `@/components/ui/link`, which should internally use `Link` from `@/i18n/routing` for proper locale handling. See the [Link component](/docs/ui/components/link) for implementation details.

```tsx
"use client";

import { Link, type LinkProps } from "@/components/ui/link";
import { useSelectedLayoutSegment } from "next/navigation";
import { ArrowUpRightIcon } from "lucide-react";

export type NavLinkProps = LinkProps & {
  showExternalIcon?: boolean;
  matchNested?: boolean;
};

export function NavLink({
  children,
  href,
  target,
  rel,
  showExternalIcon = false,
  matchNested = false,
  ...props
}: NavLinkProps) {
  const selectedLayoutSegment = useSelectedLayoutSegment();
  const pathname = selectedLayoutSegment ? `/${selectedLayoutSegment}` : "/";
  const hrefString = typeof href === "string" ? href : (href.pathname ?? "");
  const isExternal = hrefString.startsWith("http");

  const isCurrent = matchNested
    ? pathname === hrefString || pathname.startsWith(`${hrefString}/`)
    : pathname === hrefString;

  return (
    <Link
      {...props}
      href={href}
      aria-current={isCurrent ? "page" : undefined}
      data-current={isCurrent ? "true" : undefined}
      data-external={isExternal ? "true" : undefined}
      target={isExternal ? target || "_blank" : target}
      rel={isExternal ? rel || "noopener noreferrer" : rel}
    >
      {children}
      {isExternal && showExternalIcon && (
        <ArrowUpRightIcon aria-hidden="true" className="ml-1 inline size-[1em] opacity-50" />
      )}
    </Link>
  );
}
```
