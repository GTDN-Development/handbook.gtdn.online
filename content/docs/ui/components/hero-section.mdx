## Usage

```tsx
import { Button } from "@/components/ui/button";
import {
  Hero,
  HeroBackground,
  HeroBackgroundPatternGrid,
  HeroContent,
  HeroTitle,
  HeroDescription,
  HeroActions
} from "@/components/ui/hero";
```

```tsx
<Hero>
  <HeroBackground>
    <HeroBackgroundPatternGrid />
  </HeroBackground>
  <HeroContent>
    <HeroTitle>Main headline of the page</HeroTitle>
    <HeroDescription>Lorem ipsum dolor sit amet</HeroDescription>
    <HeroActions>
      <Button size="lg">Primary action</Button>
      <Button size="lg" variant="secondary">Secondary action</Button>
    </HeroActions>
  </HeroContent>
</Hero>
```

## Examples

### Without any background

By removing the `HeroBackground` component and all its content, we can have a really simple and minimal hero section.

```tsx
import { Button } from "@/components/ui/button";
import {
  Hero,
  HeroBackground,
  HeroContent,
  HeroTitle,
  HeroDescription,
  HeroActions
} from "@/components/ui/hero";
```

```tsx
export function HeroWithImageBackground() {
  return (
    <Hero>
      <HeroContent>
        <HeroTitle>Main headline of the page</HeroTitle>
        <HeroDescription>Lorem ipsum dolor sit amet</HeroDescription>
        <HeroActions>
          <Button size="lg">Primary action</Button>
          <Button size="lg" variant="secondary">Secondary action</Button>
        </HeroActions>
      </HeroContent>
    </Hero>
  );
}
```

### With Image Background

Use the Next.js `Image` component or a standard `img` element as a background with proper styling for overlay effects.

```tsx
import Image from "next/image";
import { Button } from "@/components/ui/button";
import {
  Hero,
  HeroBackground,
  HeroContent,
  HeroTitle,
  HeroDescription,
  HeroActions
} from "@/components/ui/hero";
```

```tsx
export function HeroWithImageBackground() {
  return (
    <Hero>
      <HeroBackground>
        <Image
          src="/hero-image.jpg"
          alt="hero image alt"
          width={1920}
          height={1200}
          priority
          className="size-full object-cover opacity-50"
        />
      </HeroBackground>
      <HeroContent>
        <HeroTitle>Main headline of the page</HeroTitle>
        <HeroDescription>Lorem ipsum dolor sit amet</HeroDescription>
        <HeroActions>
          <Button size="lg">Primary action</Button>
          <Button size="lg" variant="secondary">Secondary action</Button>
        </HeroActions>
      </HeroContent>
    </Hero>
  );
}
```

You can mute the image by adding some opacity to the className. There is no need to create another div as an overlay or anything else.
