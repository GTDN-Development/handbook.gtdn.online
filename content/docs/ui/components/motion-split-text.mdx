## API Reference

The component extends `<motion.div>` and adds the following props on top.

<TypeTable
  type={{
    children: {
      description: "The text content to be split and animated",
      type: "React.ReactNode",
      required: true,
    },
    split: {
      description: "Defines how the text should be split for animation",
      type: '"chars" | "words" | "both"',
      default: '"both"',
    },
    preset: {
      description: "Predefined animation preset to use",
      type: '"fade" | "slideUp" | "typing"',
      default: '"slideUp"',
    },
    containerVariants: {
      description: "Custom Framer Motion variants for the container element",
      type: "Variants",
    },
    wordVariants: {
      description: "Custom Framer Motion variants for word animations",
      type: "Variants",
    },
    charVariants: {
      description: "Custom Framer Motion variants for character animations",
      type: "Variants",
    },
  }}
/>
