import { Step, Steps } from "fumadocs-ui/components/steps";


The `cn` function combines two powerful utilities:

1. **`clsx`**: Handles conditional class names and various input formats
2. **`twMerge`**: Intelligently merges Tailwind CSS classes, resolving conflicts


## Usage
```tsx
import { cn } from "@/lib/utils";

function YourFunction({ className }) {
  return (
    <div
      className={cn(
        "flex items-start gap-2",
        className
      )}
    >
      Lorem ipsum dolor
    </div>
  );
}
```

## Installation

<Steps>
<Step>
**Create your lib/utils.ts file**

```bash
touch src/lib/utils.ts
```
</Step>

<Step>
**Install dependencies**

```bash
npm install clsx tailwind-merge
```
</Step>

<Step>
**Copy & paste the source code to your lib/utils.ts file**

```tsx title="lib/utils.ts"
import { clsx, type ClassValue } from "clsx";
import { twMerge } from "tailwind-merge";

export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs));
}
```
</Step>
</Steps>
