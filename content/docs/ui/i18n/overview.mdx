## Basic Rules  

All routes must be in the `src/app/[locale]/` directory. Always use our custom component `<Link />`. And Server components use `getTranslations`, Client components `useTranslations`.


 ## Client Components
In client components (e.g. forms, modals) use hook `useTranslations`.

```tsx
"use client";
import { useTranslations } from "next-intl";
import { Button } from "@/components/Button";

export default function CtaButton() {
  const t = useTranslations("common.navbar");

  return <Button>{t("cta")}</Button>; 
}
```

## Server Components
In server components (e.g. pages, layouts) use async function `getTranslations`.

```tsx
import { getTranslations } from "next-intl/server";
import { Button } from "@/components/Button";

export default async function HeroSection() {
  const t = await getTranslations("common.hero");

  return (
    <div>
      <h1>{t("title")}</h1>
      <p>{t("description")}</p>
      <Button>{t("cta")}</Button>
    </div>
  );
}
```

## Components for language switching

## Locale Select Component

<Steps>
<Step>
**Install required shadcn components**

This component requires the [Select](https://ui.shadcn.com/docs/components/select) component from shadcn/ui. Install it if you haven't already:

```bash
npx shadcn@latest add select
```
</Step>
<Step>
**Create the component file**
```bash
touch src/components/ui/locale-select.tsx
```
</Step>
<Step>
**Copy & paste the source code**

```tsx
"use client";

import { routing } from "@/i18n/routing";
import { useLocale } from "next-intl";
import { Locale, usePathname, useRouter } from "@/i18n/routing";
import { useParams } from "next/navigation";
import { useTransition } from "react";
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@/components/ui/select";
import clsx from "clsx";

export default function LocaleSelect({
  className = "",
}: {
  className?: string;
}) {
  const locale = useLocale();
  const router = useRouter();
  const [isPending, startTransition] = useTransition();
  const pathname = usePathname();
  const params = useParams();

  function onValueChange(nextLocale: string) {
    startTransition(() => {
      router.replace(
        // @ts-expect-error -- TypeScript will validate that only known `params`
        // are used in combination with a given `pathname`. Since the two will
        // always match for the current route, we can skip runtime checks.
        { pathname, params },
        { locale: nextLocale as Locale },
      );
    });
  }

  return (
    <Select value={locale} onValueChange={onValueChange} disabled={isPending}>
      <SelectTrigger className={clsx("w-fit", className)}>
        <SelectValue />
      </SelectTrigger>
      <SelectContent>
        {routing.locales.map((cur) => (
          <SelectItem key={cur} value={cur}>
            {cur}
          </SelectItem>
        ))}
      </SelectContent>
    </Select>
  );
}
```
</Step>

</Steps>


## Locale Button
<Steps>
<Step>
**Install required shadcn components**

This component requires the [Button](https://ui.shadcn.com/docs/components/button) and [Dropdown Menu](https://ui.shadcn.com/docs/components/dropdown-menu) components from shadcn/ui. Install them if you haven't already:

```bash
npx shadcn@latest add button dropdown-menu
```
</Step>
<Step>
**Create the component file**

```bash
touch src/components/ui/locale-button.tsx
```
</Step>
<Step>
**Copy & paste the source code**

```tsx
"use client";

import { routing } from "@/i18n/routing";
import { useLocale } from "next-intl";
import { Locale, usePathname, useRouter } from "@/i18n/routing";
import { useParams } from "next/navigation";
import { useTransition } from "react";
import { Button } from "@/components/ui/button";
import {
  DropdownMenu,
  DropdownMenuContent,
  DropdownMenuItem,
  DropdownMenuTrigger,
} from "@/components/ui/dropdown-menu";
import clsx from "clsx";

export default function LocaleSelectButton({
  className = "",
}: {
  className?: string;
}) {
  const locale = useLocale();
  const router = useRouter();
  const [isPending, startTransition] = useTransition();
  const pathname = usePathname();
  const params = useParams();

  function onSelectLocale(nextLocale: Locale) {
    startTransition(() => {
      router.replace(
        // @ts-expect-error -- TypeScript will validate that only known `params`
        // are used in combination with a given `pathname`. Since the two will
        // always match for the current route, we can skip runtime checks.
        { pathname, params },
        { locale: nextLocale },
      );
    });
  }

  // Get the display text for current locale (e.g., "CS", "EN", etc.)
  const currentLocaleDisplay = locale.toUpperCase();

  return (
    <DropdownMenu>
      <DropdownMenuTrigger asChild>
        <Button
          variant="outline"
          size="icon"
          aria-label={`Select language: ${currentLocaleDisplay}`}
          disabled={isPending}
          className={clsx("uppercase", className)}
        >
          {currentLocaleDisplay}
        </Button>
      </DropdownMenuTrigger>
      <DropdownMenuContent align="end">
        {routing.locales.map((cur) => (
          <DropdownMenuItem
            key={cur}
            onClick={() => onSelectLocale(cur)}
            className="uppercase cursor-pointer"
          >
            {cur}
          </DropdownMenuItem>
        ))}
      </DropdownMenuContent>
    </DropdownMenu>
  );
}
```
</Step>
</Steps>
